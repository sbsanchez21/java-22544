SELECT MULTIPLES TABLAS:
------------------------

Mostrar las materias del profes id=2
------------------------------------
SELECT profesores.id, profesores.nombre, profesores.apellido, materias.id, materias.nombre FROM profesores, materias WHERE profesores.id=2 AND profesores.id=materias.idProfesor 

Equivalente a la consulta anterior Join
----------------------
SELECT profesores.id, profesores.nombre, profesores.apellido, materias.id,materias.nombre FROM profesores INNER JOIN materias ON profesores.id=materias.idProfesor 


Mostrar profes y sus materias
------------------------------
SELECT profesores.id, profesores.nombre, profesores.apellido, materias.id, materias.nombre 
FROM profesores, materias 
WHERE profesores.id=materias.idProfesor


Mostrar alumno id=1 y las materias que cursa
-----------------------------
SELECT alumnos.id, alumnos.nombre, alumnos.apellido, alumnosmaterias.idMateria, materias.nombre 
FROM alumnos, alumnosmaterias, materias 
WHERE alumnos.id=1 AND alumnos.id=alumnosmaterias.idAlumno AND alumnosmaterias.idMateria=materias.id 


Mostrar todos los alumnos y las materias que cursan
--------------
SELECT alumnos.id, alumnos.nombre, alumnos.apellido, alumnosmaterias.idMateria, materias.nombre
FROM alumnos, alumnosmaterias, materias
WHERE alumnos.id=alumnosmaterias.idAlumno AND alumnosmaterias.idMateria=materias.id
ORDER BY alumnos.id


Equivalente en join tres tablas
----------------------------
